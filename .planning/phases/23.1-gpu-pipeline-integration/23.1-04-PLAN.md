**Part A: cluster/23.1_integration_smoke_dispatch.sh** — wrapper that builds a 10-ppt smoke dataset, then invokes the standard `bash cluster/21_submit_pipeline.sh` against it, capturing all submitted JobIDs for the post-run audit.

EXACT structure:

```bash
#!/bin/bash
# =============================================================================
# Phase 23.1 Wave 3 Integration Smoke Dispatch
# =============================================================================
# Prepares a 10-participant smoke dataset, then dispatches the full Phase 21
# pipeline via the GPU orchestrator (default of cluster/21_submit_pipeline.sh).
# Captures all 9 SLURM JobIDs for the post-run audit.
#
# This is the de-risk for Phase 24 cold-start: if the integrated pipeline
# breaks at smoke scale, we catch it in ~1-2h wall instead of 24-48h.
#
# Usage:
#   bash cluster/23.1_integration_smoke_dispatch.sh
#   USE_CPU=1 bash cluster/23.1_integration_smoke_dispatch.sh   # CPU dispatch dry-run
#
# Outputs:
#   output/bayesian/_smoke_data_10ppts_integration.csv  (smoke data input)
#   .planning/phases/23.1-gpu-pipeline-integration/23.1-04-jobids.txt  (JobID inventory)
# =============================================================================

set -euo pipefail
cd "$(dirname "$0")/.."

SMOKE_DATA_OUT="output/bayesian/_smoke_data_10ppts_integration.csv"
JOBIDS_FILE=".planning/phases/23.1-gpu-pipeline-integration/23.1-04-jobids.txt"
N_PPTS_LIMIT=10

mkdir -p output/bayesian "$(dirname "$JOBIDS_FILE")"

# Build the 10-ppt smoke dataset from the canonical task_trials_long.csv
echo "[$(date)] Preparing 10-participant smoke dataset"
python -c "
import pandas as pd, pathlib
df = pd.read_csv('output/task_trials_long.csv')
# Canonical participant ID column is 'sona_id' (see scripts/fitting/fit_bayesian.py
# lines 127, 129, 140, 141, 148, 152, 153, 157, 180, 183). No fallback needed.
ids = sorted(df['sona_id'].unique())[:$N_PPTS_LIMIT]
out = df[df['sona_id'].isin(ids)].copy()
out.to_csv('$SMOKE_DATA_OUT', index=False)
print(f'Wrote {len(out)} rows for {len(ids)} unique participants to $SMOKE_DATA_OUT')
"

echo ""
echo "[$(date)] Dispatching Phase 21 pipeline via orchestrator"
echo "  Mode: ${USE_CPU:+CPU (USE_CPU=1)}${USE_CPU:-GPU (default)}"
echo ""

# CRITICAL: the SLURM scripts hard-code --data output/task_trials_long.csv.
# To use the smoke dataset without rewriting all SLURMs, do a temporary
# symlink swap. Save the original; symlink smoke -> task_trials_long.csv;
# run pipeline; restore on exit (trap).
ORIG_DATA="output/task_trials_long.csv"
ORIG_DATA_BACKUP="output/task_trials_long.csv.pre_23_1_smoke_backup"

cleanup() {
    if [[ -f "$ORIG_DATA_BACKUP" && ! -L "$ORIG_DATA" ]]; then
        : # original was a regular file, not symlinked; nothing to restore
    elif [[ -f "$ORIG_DATA_BACKUP" ]]; then
        echo "[$(date)] Restoring original $ORIG_DATA"
        rm -f "$ORIG_DATA"
        mv "$ORIG_DATA_BACKUP" "$ORIG_DATA"
    fi
}
trap cleanup EXIT

if [[ -f "$ORIG_DATA" && ! -L "$ORIG_DATA" ]]; then
    cp "$ORIG_DATA" "$ORIG_DATA_BACKUP"
fi
cp "$SMOKE_DATA_OUT" "$ORIG_DATA"

# Now dispatch — orchestrator reads from output/task_trials_long.csv (which
# is now the smoke 10-ppt copy). USE_CPU env var passes through to orchestrator.
bash cluster/21_submit_pipeline.sh 2>&1 | tee "$JOBIDS_FILE.log"

# Extract JobIDs from the orchestrator stdout.
# NOTE: step 21.2 echoes TWO JobIDs on a single line:
#   "  [21.2] $m -> array job ${REC_ARRAY[$m]}, aggregate ${REC_AGG[$m]}"
# A greedy sed like 's/.*-> (...)?([0-9]+).*/\2/' only captures the LAST number per
# line — dropping the array JobID and silently letting failed array tasks escape
# the sacct audit. Use Python multi-match (re.findall) to capture every 6+-digit
# number on every orchestrator echo line. This picks up BOTH JobIDs from the
# 21.2 dual-JobID line.
python -c "
import re, pathlib
log = pathlib.Path('$JOBIDS_FILE.log').read_text()
jobids = []
for line in log.splitlines():
    # Match every orchestrator echo line: '  [21.N]' or '  [21.N.letter]'
    if re.search(r'^\s*\[21\.[0-9]+(\.[a-z]+)?\]', line):
        jobids.extend(re.findall(r'\b(\d{6,})\b', line))
# Deduplicate, preserving order
seen = set()
unique_ids = [j for j in jobids if not (j in seen or seen.add(j))]
pathlib.Path('$JOBIDS_FILE').write_text('\n'.join(unique_ids) + '\n')
print(f'Captured {len(unique_ids)} unique JobIDs to $JOBIDS_FILE')
"

echo ""
echo "[$(date)] Pipeline dispatched. JobIDs captured to $JOBIDS_FILE"
echo "  Monitor: squeue -u \$USER --jobs=\$(tr '\\n' ',' < $JOBIDS_FILE | sed 's/,\$//')"
echo "  Post-run audit: python validation/check_phase_23_1_integration.py"
```
