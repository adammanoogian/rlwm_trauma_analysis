7. The model loop (PER-MODEL timeout pattern from cluster/13_bayesian_fullybatched_smoke.slurm lines 121-165):
```
declare -A RESULTS
OVERALL_PASS=true
for MODEL in $MODELS; do
    echo ""
    echo "------------------------------------------------------------"
    echo "  $MODEL (timeout ${PER_MODEL_TIMEOUT}s)"
    echo "  Started: $(date)"
    echo "------------------------------------------------------------"
    MODEL_START=$(date +%s)
    timeout "$PER_MODEL_TIMEOUT" python scripts/fitting/fit_bayesian.py \
        --model "$MODEL" \
        --data "$SMOKE_DATA" \
        --output "$SMOKE_OUT_DIR" \
        --chains "$CHAINS" \
        --warmup "$WARMUP" \
        --samples "$SAMPLES" \
        --max-tree-depth "$MAX_TREE_DEPTH" \
        --seed 42
    MODEL_EXIT=$?
    MODEL_END=$(date +%s)
    MODEL_WALL=$((MODEL_END - MODEL_START))
    if [[ $MODEL_EXIT -eq 0 ]]; then
        RESULTS[$MODEL]="PASS (${MODEL_WALL}s)"
    elif [[ $MODEL_EXIT -eq 124 ]]; then
        RESULTS[$MODEL]="TIMEOUT (${MODEL_WALL}s, exceeded ${PER_MODEL_TIMEOUT}s)"; OVERALL_PASS=false
    else
        RESULTS[$MODEL]="EXIT=$MODEL_EXIT (${MODEL_WALL}s)"
        if [[ $MODEL_WALL -lt 30 ]]; then OVERALL_PASS=false; fi
    fi
    echo "  $MODEL: exit=$MODEL_EXIT, wall=${MODEL_WALL}s"
done
```
