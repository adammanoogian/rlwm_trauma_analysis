    Create `scripts/fitting/numpyro_helpers.py` with the following API:

    ```python
    """Non-centered parameterization helpers for NumPyro hierarchical models.

    Implements the hBayesDM convention (Ahn, Haines, Zhang 2017):
    theta_unc = mu_pr + sigma_pr * z
    theta = Phi_approx(theta_unc) * upper_bound + lower_bound

    Where Phi_approx is the standard normal CDF (jax.scipy.stats.norm.cdf).

    v4.0 INFRA-05.
    """
    from __future__ import annotations
    ```

    **Functions to implement:**

    1. `phi_approx(x: jnp.ndarray) -> jnp.ndarray`:
       - Wraps `jax.scipy.stats.norm.cdf(x)`.
       - One-liner, but named for API clarity and grep-ability.

    2. `sample_bounded_param(name: str, *, lower: float = 0.0, upper: float = 1.0, n_participants: int, mu_prior_loc: float = 0.0, mu_prior_scale: float = 1.0, sigma_prior_scale: float = 0.2) -> jnp.ndarray`:
       - Samples a non-centered bounded parameter for `n_participants`.
       - Implementation:
         ```python
         mu_pr = numpyro.sample(f"{name}_mu_pr", dist.Normal(mu_prior_loc, mu_prior_scale))
         sigma_pr = numpyro.sample(f"{name}_sigma_pr", dist.HalfNormal(sigma_prior_scale))
         z = numpyro.sample(f"{name}_z", dist.Normal(0, 1).expand([n_participants]))
         theta_unc = mu_pr + sigma_pr * z
         theta = lower + (upper - lower) * phi_approx(theta_unc)
         numpyro.deterministic(name, theta)
         return theta
         ```
       - This single function handles ALL [0,1] bounded parameters (alpha_pos, alpha_neg, epsilon, phi, rho, kappa, kappa_s, kappa_total, kappa_share, phi_rl) by varying `lower`, `upper`, and `mu_prior_loc`.

    3. `sample_capacity(name: str = "capacity", *, n_participants: int, lower: float = 2.0, upper: float = 6.0, mu_prior_loc: float = 0.0, mu_prior_scale: float = 1.0, sigma_prior_scale: float = 0.2) -> jnp.ndarray`:
       - Convenience wrapper for K with Collins bounds.
       - Calls `sample_bounded_param(name, lower=lower, upper=upper, n_participants=n_participants, mu_prior_loc=mu_prior_loc, mu_prior_scale=mu_prior_scale, sigma_prior_scale=sigma_prior_scale)`.
       - The separate function exists for API clarity and to lock the K defaults.

    4. `PARAM_PRIOR_DEFAULTS: dict[str, dict]`:
       - A dictionary mapping parameter names to their default prior settings:
         ```python
         PARAM_PRIOR_DEFAULTS = {
             "alpha_pos": {"lower": 0.0, "upper": 1.0, "mu_prior_loc": 0.0},
             "alpha_neg": {"lower": 0.0, "upper": 1.0, "mu_prior_loc": 0.0},
             "epsilon": {"lower": 0.0, "upper": 1.0, "mu_prior_loc": -2.5},
             "phi": {"lower": 0.0, "upper": 1.0, "mu_prior_loc": -0.8},

             "kappa_s": {"lower": 0.0, "upper": 1.0, "mu_prior_loc": -2.0},

         }
         ```
       - Phase 15+ models will use these defaults via `sample_bounded_param(name, **PARAM_PRIOR_DEFAULTS[name], n_participants=N)`.

    ```python
    # ==========================================================================
    # Parameterization Version Registry (v4.0 INFRA-06)
    # ==========================================================================

    EXPECTED_PARAMETERIZATION: dict[str, str] = {
        "qlearning": "v4.0-phiapprox",
        "wmrl": "v4.0-K[2,6]-phiapprox",
        "wmrl_m3": "v4.0-K[2,6]-phiapprox",
        "wmrl_m5": "v4.0-K[2,6]-phiapprox",
        "wmrl_m6a": "v4.0-K[2,6]-phiapprox",
        "wmrl_m6b": "v4.0-K[2,6]-phiapprox-stickbreaking",
        "wmrl_m4": "v4.0-K[2,6]-phiapprox-lba",
    }


    def load_fits_with_validation(
        path: Path, model: str
    ) -> "pd.DataFrame":
        """Load a fit CSV and validate its parameterization_version.

        Parameters
        ----------
        path : Path
            Path to the CSV file.
        model : str
            Model name key into EXPECTED_PARAMETERIZATION.

        Returns
        -------
        pd.DataFrame
            The loaded DataFrame.

        Raises
        ------
        ValueError
            If parameterization_version column is missing or mismatched.
            Includes expected vs actual values in error message.
        """
        import pandas as pd

        df = pd.read_csv(path)
        if "parameterization_version" not in df.columns:
            raise ValueError(
                f"{path} lacks 'parameterization_version' column — "
                f"likely a v3.0 legacy fit. Expected: {EXPECTED_PARAMETERIZATION.get(model, 'unknown')}"
            )
        expected = EXPECTED_PARAMETERIZATION[model]
        actual = df["parameterization_version"].unique()
        if len(actual) != 1 or actual[0] != expected:
            raise ValueError(
                f"{path} parameterization_version mismatch: "
                f"expected='{expected}', actual={list(actual)}"
            )
        return df
    ```
