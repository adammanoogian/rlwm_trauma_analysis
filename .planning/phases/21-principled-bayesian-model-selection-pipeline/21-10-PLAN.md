Create `cluster/21_submit_pipeline.sh` — the master orchestrator chaining all 9 pipeline steps. Adapt from `cluster/submit_full_pipeline.sh`.

```bash
#!/bin/bash
# =============================================================================
# Phase 21: Principled Bayesian Model Selection Pipeline — Master Orchestrator
# =============================================================================
# Chains all 9 pipeline steps via sbatch --dependency=afterok.
# Exit code 2 from step 21.5 (inconclusive winner) pauses the chain; user
# reviews output/bayesian/21_baseline/winner_report.md and resubmits steps
# 21.6+ manually with --force-winners.
# Exit code 2 from step 21.7 (NULL_RESULT) auto-skips 21.8 and proceeds to 21.9.
#
# Pre-flight gate (plan 21-11): the 2-covariate L2 hook for M3/M5/M6a lives
# in scripts/fitting/numpyro_models.py. It MUST pass its fast pytest suite
# before any cluster job is submitted. This is a local gate, not a SLURM job.
# =============================================================================
set -euo pipefail
cd "$(dirname "$0")/.."

MODELS="qlearning wmrl wmrl_m3 wmrl_m5 wmrl_m6a wmrl_m6b"

echo "[$(date)] === Phase 21 pipeline orchestrator ==="

# ---- Pre-flight gate: 2-covariate L2 infrastructure (plan 21-11) ----
echo "[$(date)] Pre-flight: verifying 2-covariate L2 hook (plan 21-11)..."
pytest scripts/fitting/tests/test_numpyro_models_2cov.py -v -k "not slow" --tb=short || {
  echo "[ABORT] plan 21-11 tests failed — 2-covariate L2 hook for M3/M5/M6a is not wired correctly."
  echo "[ABORT] No cluster jobs submitted. Fix numpyro_models.py and retry."
  exit 1
}
echo "[$(date)] Pre-flight OK — 2-covariate L2 hook ready."

# Step 21.1 — prior predictive (6 parallel)
declare -A PPC
for m in $MODELS; do
  PPC[$m]=$(sbatch --parsable --export=ALL,MODEL=$m cluster/21_1_prior_predictive.slurm)
  echo "  [21.1] $m → job ${PPC[$m]}"
done
PPC_DEP="afterok:$(IFS=:; echo "${PPC[*]}")"

# Step 21.2 — recovery (6 models × 50 array tasks + 6 aggregation steps, parallel)
declare -A REC_ARRAY
declare -A REC_AGG
for m in $MODELS; do
  REC_ARRAY[$m]=$(sbatch --parsable --dependency=afterok:${PPC[$m]} \
    --export=ALL,MODEL=$m cluster/21_2_recovery.slurm)
  REC_AGG[$m]=$(sbatch --parsable --dependency=afterok:${REC_ARRAY[$m]} \
    --export=ALL,MODEL=$m cluster/21_2_recovery_aggregate.slurm)
  echo "  [21.2] $m → array job ${REC_ARRAY[$m]}, aggregate ${REC_AGG[$m]}"
done
REC_DEP="afterok:$(IFS=:; echo "${REC_AGG[*]}")"

# Step 21.3 — baseline hierarchical fits (6 parallel)
declare -A BASE
for m in $MODELS; do
  BASE[$m]=$(sbatch --parsable --dependency=${REC_DEP} \
    --export=ALL,MODEL=$m cluster/21_3_fit_baseline.slurm)
  echo "  [21.3] $m → job ${BASE[$m]}"
done
BASE_DEP="afterok:$(IFS=:; echo "${BASE[*]}")"

# Step 21.4 — convergence audit (1 job, after all baseline fits)
AUDIT_JID=$(sbatch --parsable --dependency=${BASE_DEP} cluster/21_4_baseline_audit.slurm)
echo "  [21.4] convergence audit → job $AUDIT_JID"

# Step 21.5 — LOO + stacking + BMS (1 job, after audit passes)
LOO_JID=$(sbatch --parsable --dependency=afterok:$AUDIT_JID cluster/21_5_loo_stacking_bms.slurm)
echo "  [21.5] LOO+stacking+BMS → job $LOO_JID"

# Step 21.6 — fit winner(s) with L2 (dynamic: one job per winner, submitted manually
#             after user reviews winner_report.md OR automatically if winners.txt is
#             written and exit code was 0)
# Orchestrator cannot read winners.txt before step 21.5 runs. So we submit a single
# post-processing job that re-invokes sbatch per winner.
# NOTE (plan-checker Issue #6): the dispatcher is wrapped in a proper SLURM job
# (cluster/21_6_dispatch_l2.slurm, 14h time cap) because the dispatcher uses
# `sbatch --wait` internally and may block for up to 12h (M6b subscale worst case).
# A `--wrap=` job inherits short default time limits and would be killed by the
# scheduler long before the L2 fits complete.
DISPATCH_JID=$(sbatch --parsable --dependency=afterok:$LOO_JID \
    cluster/21_6_dispatch_l2.slurm)
echo "  [21.6] L2 dispatch (reads winners.txt) → job $DISPATCH_JID"

# Step 21.7 — scale audit (depends on all L2 winner fits; dispatch job will write
#             a completion flag that this dependency watches)
# Simpler: scale audit depends on the dispatcher's exit, and the dispatcher blocks
# until each child L2 fit finishes (use --wait in dispatcher).
AUDIT2_JID=$(sbatch --parsable --dependency=afterok:$DISPATCH_JID cluster/21_7_scale_audit.slurm)
echo "  [21.7] scale audit → job $AUDIT2_JID"

# Step 21.8 — model averaging
# NOTE (plan-checker Issue #4 resolution): Use afterok — NOT afterany — because
# plan 21-08's scale audit was updated to exit 0 for both PROCEED_TO_AVERAGING and
# NULL_RESULT (distinguished by the YAML pipeline_action header inside the audit
# report). Exit 1 is reserved for genuine audit errors, in which case afterok
# correctly blocks averaging. 21_model_averaging.py reads the YAML header itself
# and soft-skips averaging when pipeline_action == NULL_RESULT.
AVG_JID=$(sbatch --parsable --dependency=afterok:$AUDIT2_JID cluster/21_8_model_averaging.slurm)
echo "  [21.8] averaging → job $AVG_JID"

# Step 21.9 — manuscript tables (final)
TABLES_JID=$(sbatch --parsable --dependency=afterok:$AVG_JID cluster/21_9_manuscript_tables.slurm)
echo "  [21.9] tables → job $TABLES_JID"

echo "[$(date)] === All jobs submitted ==="
echo "Monitor progress: squeue -u \$USER"
echo "Final output: output/bayesian/21_tables/"
```

#!/bin/bash
set -euo pipefail
cd "$(dirname "$0")/.."
WINNERS_FILE=output/bayesian/21_baseline/winners.txt
if [ ! -f "$WINNERS_FILE" ]; then
  echo "[DISPATCH] No winners.txt — step 21.5 may have exited with code 2; pipeline paused."
  exit 2
fi
WINNERS=$(cat "$WINNERS_FILE" | tr ',' ' ')
declare -A NAME_MAP=( ["M1"]="qlearning" ["M2"]="wmrl" ["M3"]="wmrl_m3" \
                       ["M5"]="wmrl_m5" ["M6a"]="wmrl_m6a" ["M6b"]="wmrl_m6b" )
for name in $WINNERS; do
  model=${NAME_MAP[$name]:-$name}
  echo "[DISPATCH] Submitting L2 fit for $name ($model) and blocking on --wait"

done

echo "[DISPATCH] All L2 winner fits complete; releasing downstream afterok dependency"

The dispatcher itself stays alive until all children complete — this is why `cluster/21_6_dispatch_l2.slurm` has `--time=14:00:00` (plan-checker Issue #6). The master orchestrator's `afterok:$DISPATCH_JID` dependency releases step 21.7 only after the dispatcher exits cleanly.
