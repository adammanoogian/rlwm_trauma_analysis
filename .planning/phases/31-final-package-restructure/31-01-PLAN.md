```ini
# Test discovery — Phase 31 consolidated layout (tests/{unit,integration,scientific})

testpaths = tests
python_files = test_*.py

python_functions = test_*

pythonpath = .

markers =
    slow: marks tests as slow (deselect with '-m "not slow"')
    integration: marks tests as integration tests
    scientific: marks tests as scientific-validation tests (slow, deselect with '-m "not scientific"')

# Default options
addopts =
    -v
    --strict-markers
    --tb=short
    --durations=10
```
