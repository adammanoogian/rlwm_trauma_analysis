def compute_akaike_weights_n(aic_values: Dict[str, float]) -> Dict[str, float]:
    """
    Compute Akaike weights for N models.

    w_i = exp(-0.5 * delta_i) / sum(exp(-0.5 * delta_j))

    Args:
        aic_values: Dict mapping model name to aggregate AIC

    Returns:
        Dict mapping model name to Akaike weight
    """
    min_aic = min(aic_values.values())
    deltas = {name: aic - min_aic for name, aic in aic_values.items()}

    exp_terms = {name: np.exp(-0.5 * delta) for name, delta in deltas.items()}
    total = sum(exp_terms.values())

    return {name: exp_val / total for name, exp_val in exp_terms.items()}


def compare_models(
    fits_dict: Dict[str, pd.DataFrame],
    metric: str = 'aic'
) -> pd.DataFrame:
    """
    Compare N models using AIC or BIC.

    Args:
        fits_dict: Dict mapping model name ('M1', 'M2', 'M3') to fit DataFrame
        metric: 'aic' or 'bic'

    Returns:
        DataFrame with comparison results
    """
    # Model metadata
    n_params_map = {
        'M1': 3,   # Q-learning: alpha_pos, alpha_neg, epsilon
        'M2': 6,   # WM-RL: alpha_pos, alpha_neg, phi, rho, capacity, epsilon
        'M3': 7,   # WM-RL+kappa: alpha_pos, alpha_neg, phi, rho, capacity, kappa, epsilon
    }
    model_names_full = {
        'M1': 'M1 (Q-learning)',
        'M2': 'M2 (WM-RL)',
        'M3': 'M3 (WM-RL+kappa)',
    }

    # Compute aggregate IC for each model (converged only)
    agg_ic = {}
    n_converged = {}
    for name, df in fits_dict.items():
        converged_df = df[df['converged'] == True]
        if len(converged_df) == 0:
            converged_df = df  # Fallback if none converged
        agg_ic[name] = converged_df[metric].sum()
        n_converged[name] = len(converged_df)

    # Compute deltas from best
    min_ic = min(agg_ic.values())
    deltas = {name: ic - min_ic for name, ic in agg_ic.items()}

    # Compute Akaike weights
    weights = compute_akaike_weights_n(agg_ic)

    # Build comparison DataFrame
    rows = []
    for name in fits_dict.keys():
        rows.append({
            'model': model_names_full.get(name, name),
            'n_params': n_params_map.get(name, '?'),
            'n_converged': n_converged[name],
            metric: agg_ic[name],
            f'delta_{metric}': deltas[name],
            'weight': weights[name],
            'evidence': interpret_delta_aic(deltas[name]) if metric == 'aic' else ''
        })

    return pd.DataFrame(rows)


def count_participant_wins_n(
    fits_dict: Dict[str, pd.DataFrame],
    metric: str = 'aic'
) -> pd.DataFrame:
    """
    Count which model wins for each participant (N models).

    Args:
        fits_dict: Dict mapping model name to fit DataFrame
        metric: 'aic' or 'bic'

    Returns:
        DataFrame with win counts per model
    """
    model_names = list(fits_dict.keys())

    # Build merged DataFrame
    merged = None
    for name, df in fits_dict.items():
        subset = df[['participant_id', metric, 'converged']].rename(
            columns={metric: f'{metric}_{name}', 'converged': f'conv_{name}'}
        )
        if merged is None:
            merged = subset
        else:
            merged = merged.merge(subset, on='participant_id', how='inner')

    # Filter to all converged
    conv_cols = [f'conv_{name}' for name in model_names]
    all_conv_mask = merged[conv_cols].all(axis=1)
    all_conv = merged[all_conv_mask].copy()

    if len(all_conv) == 0:
        return pd.DataFrame({'model': model_names, 'wins': [0]*len(model_names), 'percent': [0.0]*len(model_names)})

    # Determine winner per participant
    ic_cols = [f'{metric}_{name}' for name in model_names]

    def find_winner(row):
        vals = [row[col] for col in ic_cols]
        min_idx = np.argmin(vals)
        return model_names[min_idx]

    all_conv['winner'] = all_conv.apply(find_winner, axis=1)

    # Count wins
    win_counts = all_conv['winner'].value_counts()
    total = len(all_conv)

    rows = []
    for name in model_names:
        wins = win_counts.get(name, 0)
        rows.append({
            'model': name,
            'wins': wins,
            'percent': 100.0 * wins / total if total > 0 else 0.0
        })

    return pd.DataFrame(rows)

    parser = argparse.ArgumentParser(
        description='Compare MLE fits between Q-Learning (M1), WM-RL (M2), and WM-RL+kappa (M3) models'
    )
    # Legacy arguments (for backward compatibility)

    parser.add_argument('--wmrl', type=str, default=None,
                        help='Path to WM-RL (M2) individual fits CSV (legacy)')

    # New arguments for explicit model naming
    parser.add_argument('--m1', type=str, default=None,
                        help='Path to M1 (Q-Learning) individual fits CSV')
    parser.add_argument('--m2', type=str, default=None,
                        help='Path to M2 (WM-RL) individual fits CSV')
    parser.add_argument('--m3', type=str, default=None,
                        help='Path to M3 (WM-RL+kappa) individual fits CSV')

    parser.add_argument('--output', type=str, default=None,
                        help='Output path for comparison results (optional)')

    args = parser.parse_args()

    # Handle legacy argument names
    m1_path = args.m1 or args.qlearning
    m2_path = args.m2 or args.wmrl
    m3_path = args.m3

    # Build fits dictionary from available models
    fits_dict = {}
    if m1_path:
        fits_dict['M1'] = load_fits(m1_path)
        print(f"Loaded M1 (Q-Learning): {len(fits_dict['M1'])} participants")
    if m2_path:
        fits_dict['M2'] = load_fits(m2_path)
        print(f"Loaded M2 (WM-RL): {len(fits_dict['M2'])} participants")
    if m3_path:
        fits_dict['M3'] = load_fits(m3_path)
        print(f"Loaded M3 (WM-RL+kappa): {len(fits_dict['M3'])} participants")

    if len(fits_dict) < 2:
        parser.error("At least 2 model fits required for comparison")
