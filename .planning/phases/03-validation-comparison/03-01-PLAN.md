"""
Backward Compatibility Tests for WM-RL M3 Model

Validates that M3 with kappa=0 produces numerically identical
likelihood values to M2. This ensures the perseveration extension
does not break baseline behavior.

Following Senta et al. (2025) methodology:
- M2: WM-RL hybrid (alpha_pos, alpha_neg, phi, rho, capacity, epsilon)
- M3: WM-RL + kappa (alpha_pos, alpha_neg, phi, rho, capacity, kappa, epsilon)

When kappa=0, M3 should exactly match M2.
"""

import pytest
import jax
import jax.numpy as jnp

from scripts.fitting.jax_likelihoods import (
    wmrl_block_likelihood,
    wmrl_m3_block_likelihood,
    wmrl_multiblock_likelihood,
    wmrl_m3_multiblock_likelihood,
)


# =============================================================================
# Fixtures
# =============================================================================

@pytest.fixture
def shared_params():
    """M2 parameters (shared between M2 and M3 tests)."""
    return {
        'alpha_pos': 0.3,
        'alpha_neg': 0.1,
        'phi': 0.1,
        'rho': 0.7,
        'capacity': 4.0,
        'epsilon': 0.05
    }


@pytest.fixture
def generate_block_data():
    """Factory fixture for generating random block data."""
    def _generate(seed: int, n_trials: int = 50):
        key = jax.random.PRNGKey(seed)
        stimuli = jax.random.randint(key, (n_trials,), 0, 6)
        key, subkey = jax.random.split(key)
        actions = jax.random.randint(subkey, (n_trials,), 0, 3)
        key, subkey = jax.random.split(key)
        rewards = jax.random.bernoulli(subkey, 0.7, (n_trials,)).astype(jnp.float32)
        set_sizes = jnp.ones((n_trials,)) * 5
        return stimuli, actions, rewards, set_sizes
    return _generate


# =============================================================================
# Single Block Tests
# =============================================================================

class TestSingleBlockBackwardCompatibility:
    """Test backward compatibility for single-block likelihood."""

    def test_m3_kappa0_matches_m2(self, shared_params, generate_block_data):
        """M3 with kappa=0 should exactly match M2 (single block)."""
        stimuli, actions, rewards, set_sizes = generate_block_data(seed=42)

        # M2 likelihood
        log_lik_m2 = wmrl_block_likelihood(
            stimuli, actions, rewards, set_sizes, **shared_params
        )

        # M3 with kappa=0
        log_lik_m3 = wmrl_m3_block_likelihood(
            stimuli, actions, rewards, set_sizes, **shared_params, kappa=0.0
        )

        # Strict numerical equivalence
        assert jnp.allclose(log_lik_m2, log_lik_m3, rtol=1e-5, atol=1e-8), \
            f"Backward compatibility failed: M2={log_lik_m2:.8f}, M3(kappa=0)={log_lik_m3:.8f}"

    @pytest.mark.parametrize("seed", [42, 123, 456, 789, 1024])
    def test_m3_kappa0_matches_m2_multiple_seeds(self, shared_params, seed):
        """Test backward compatibility across multiple random seeds."""
        key = jax.random.PRNGKey(seed)
        n_trials = 30

        stimuli = jax.random.randint(key, (n_trials,), 0, 6)
        key, subkey = jax.random.split(key)
        actions = jax.random.randint(subkey, (n_trials,), 0, 3)
        key, subkey = jax.random.split(key)
        rewards = jax.random.bernoulli(subkey, 0.6, (n_trials,)).astype(jnp.float32)
        set_sizes = jnp.ones((n_trials,)) * 4

        log_lik_m2 = wmrl_block_likelihood(
            stimuli, actions, rewards, set_sizes, **shared_params
        )
        log_lik_m3 = wmrl_m3_block_likelihood(
            stimuli, actions, rewards, set_sizes, **shared_params, kappa=0.0
        )

        assert jnp.allclose(log_lik_m2, log_lik_m3, rtol=1e-5, atol=1e-8), \
            f"Seed {seed} failed: M2={log_lik_m2:.8f}, M3(kappa=0)={log_lik_m3:.8f}"

    @pytest.mark.parametrize("params_variant", [
        {'alpha_pos': 0.1, 'alpha_neg': 0.05, 'phi': 0.2, 'rho': 0.5, 'capacity': 3.0, 'epsilon': 0.1},
        {'alpha_pos': 0.5, 'alpha_neg': 0.3, 'phi': 0.05, 'rho': 0.9, 'capacity': 6.0, 'epsilon': 0.02},
        {'alpha_pos': 0.8, 'alpha_neg': 0.8, 'phi': 0.5, 'rho': 0.3, 'capacity': 2.0, 'epsilon': 0.15},
    ])
    def test_m3_kappa0_matches_m2_various_params(self, params_variant, generate_block_data):
        """Test backward compatibility with various parameter combinations."""
        stimuli, actions, rewards, set_sizes = generate_block_data(seed=999)

        log_lik_m2 = wmrl_block_likelihood(
            stimuli, actions, rewards, set_sizes, **params_variant
        )
        log_lik_m3 = wmrl_m3_block_likelihood(
            stimuli, actions, rewards, set_sizes, **params_variant, kappa=0.0
        )

        assert jnp.allclose(log_lik_m2, log_lik_m3, rtol=1e-5, atol=1e-8), \
            f"Failed with params {params_variant}"


# =============================================================================
# Multi-Block Tests
# =============================================================================

class TestMultiBlockBackwardCompatibility:
    """Test backward compatibility for multi-block likelihood."""

    def test_m3_kappa0_matches_m2_multiblock(self, shared_params):
        """M3 with kappa=0 should exactly match M2 (multi-block)."""
        key = jax.random.PRNGKey(123)

        stimuli_blocks = []
        actions_blocks = []
        rewards_blocks = []
        set_sizes_blocks = []

        for i in range(3):
            key, subkey = jax.random.split(key)
            stimuli_blocks.append(jax.random.randint(subkey, (40,), 0, 6))
            key, subkey = jax.random.split(key)
            actions_blocks.append(jax.random.randint(subkey, (40,), 0, 3))
            key, subkey = jax.random.split(key)
            rewards_blocks.append(jax.random.bernoulli(subkey, 0.7, (40,)).astype(jnp.float32))
            set_sizes_blocks.append(jnp.ones((40,)) * (i + 3))

        log_lik_m2 = wmrl_multiblock_likelihood(
            stimuli_blocks, actions_blocks, rewards_blocks, set_sizes_blocks,
            **shared_params
        )
        log_lik_m3 = wmrl_m3_multiblock_likelihood(
            stimuli_blocks, actions_blocks, rewards_blocks, set_sizes_blocks,
            **shared_params, kappa=0.0
        )

        assert jnp.allclose(log_lik_m2, log_lik_m3, rtol=1e-5, atol=1e-8), \
            f"Multi-block backward compatibility failed: M2={log_lik_m2:.8f}, M3(kappa=0)={log_lik_m3:.8f}"

    def test_m3_kappa0_matches_m2_many_blocks(self, shared_params):
        """Test with realistic number of blocks (23, like real experiment)."""
        key = jax.random.PRNGKey(777)
        n_blocks = 23
        trials_per_block = 24

        stimuli_blocks = []
        actions_blocks = []
        rewards_blocks = []
        set_sizes_blocks = []

        for _ in range(n_blocks):
            key, subkey = jax.random.split(key)
            stimuli_blocks.append(jax.random.randint(subkey, (trials_per_block,), 0, 6))
            key, subkey = jax.random.split(key)
            actions_blocks.append(jax.random.randint(subkey, (trials_per_block,), 0, 3))
            key, subkey = jax.random.split(key)
            rewards_blocks.append(jax.random.bernoulli(subkey, 0.7, (trials_per_block,)).astype(jnp.float32))
            set_sizes_blocks.append(jnp.ones((trials_per_block,)) * 5)

        log_lik_m2 = wmrl_multiblock_likelihood(
            stimuli_blocks, actions_blocks, rewards_blocks, set_sizes_blocks,
            **shared_params
        )
        log_lik_m3 = wmrl_m3_multiblock_likelihood(
            stimuli_blocks, actions_blocks, rewards_blocks, set_sizes_blocks,
            **shared_params, kappa=0.0
        )

        assert jnp.allclose(log_lik_m2, log_lik_m3, rtol=1e-5, atol=1e-8), \
            f"23-block test failed: M2={log_lik_m2:.8f}, M3(kappa=0)={log_lik_m3:.8f}"


# =============================================================================
# Sanity Checks (kappa > 0 should differ)
# =============================================================================

class TestKappaEffect:
    """Verify that kappa parameter actually has an effect."""

    def test_kappa_positive_differs_from_zero(self, shared_params, generate_block_data):
        """Kappa > 0 should produce different likelihood than kappa=0."""
        stimuli, actions, rewards, set_sizes = generate_block_data(seed=456)

        log_lik_kappa0 = wmrl_m3_block_likelihood(
            stimuli, actions, rewards, set_sizes, **shared_params, kappa=0.0
        )
        log_lik_kappa_pos = wmrl_m3_block_likelihood(
            stimuli, actions, rewards, set_sizes, **shared_params, kappa=0.5
        )

        # Likelihoods should be DIFFERENT
        assert not jnp.allclose(log_lik_kappa0, log_lik_kappa_pos, rtol=1e-3), \
            f"kappa parameter has no effect! kappa=0: {log_lik_kappa0:.6f}, kappa=0.5: {log_lik_kappa_pos:.6f}"

    def test_kappa_effect_with_repetitive_actions(self, shared_params):
        """
        With deliberate action repetition, higher kappa should increase likelihood.

        This tests that perseveration is working as intended: when actions repeat,
        the model with kappa > 0 should assign higher probability.
        """
        n_trials = 50
        key = jax.random.PRNGKey(111)

        # Random stimuli
        stimuli = jax.random.randint(key, (n_trials,), 0, 6)

        # Highly repetitive actions (same action 10 times, then switch)
        actions = jnp.array([0]*10 + [1]*10 + [2]*10 + [0]*10 + [1]*10, dtype=jnp.int32)

        key, subkey = jax.random.split(key)
        rewards = jax.random.bernoulli(subkey, 0.7, (n_trials,)).astype(jnp.float32)
        set_sizes = jnp.ones((n_trials,)) * 5

        log_lik_kappa0 = wmrl_m3_block_likelihood(
            stimuli, actions, rewards, set_sizes, **shared_params, kappa=0.0
        )
        log_lik_kappa_high = wmrl_m3_block_likelihood(
            stimuli, actions, rewards, set_sizes, **shared_params, kappa=0.8
        )

        # With repetitive actions and high kappa, likelihood should be HIGHER
        # (more perseveration = higher prob of repeating)
        assert log_lik_kappa_high > log_lik_kappa0, \
            f"High kappa should increase likelihood for repetitive actions. " \
            f"kappa=0: {log_lik_kappa0:.6f}, kappa=0.8: {log_lik_kappa_high:.6f}"

    @pytest.mark.parametrize("kappa", [0.1, 0.3, 0.5, 0.7, 0.9])
    def test_various_kappa_values(self, shared_params, generate_block_data, kappa):
        """Test that various kappa values produce valid likelihoods."""
        stimuli, actions, rewards, set_sizes = generate_block_data(seed=333)

        log_lik = wmrl_m3_block_likelihood(
            stimuli, actions, rewards, set_sizes, **shared_params, kappa=kappa
        )

        # Likelihood should be finite and negative (log-probability)
        assert jnp.isfinite(log_lik), f"Non-finite likelihood for kappa={kappa}"
        assert log_lik < 0, f"Likelihood should be negative for kappa={kappa}, got {log_lik}"
